def row_to_str(row: list):
    rstr = ""
    for r in row:
        if r == 0:
            rstr += " "
        else:
            rstr += "*"
    return rstr

def canvas_to_string(canvas):
    ""
    cstr = ""
    for row in canvas:
        rstr = row_to_str(row)
        cstr += rstr + "\n"
    return cstr
